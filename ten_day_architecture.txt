```typescript
// softshape-backend\src\middleware\tenantCompat.ts
// LEGACY IDs in the existing system:
const LEGACY_IDS = new Set(['restaurant-001', 'bar-001', 'venue-001']);

export function tenantCompat(req: any, res: any, next: any) {
  const slug = req.headers['x-tenant-slug'] as string | undefined;

  if (!slug || LEGACY_IDS.has(slug)) {
    // No header OR request is for a known legacy outlet — leave req.restaurantId as-is
    req.tenantId = slug || 'restaurant-001';
    req.isLegacy = true;
  } else {
    req.tenantId = slug;
    req.isLegacy = false;
  }
  next();
}
```

```typescript
import jwt from 'jsonwebtoken';
export function jwtAuth(req: any, res: any, next: any) {
  const token = req.headers['authorization']?.replace('Bearer ', '');
  if (!token) return res.status(401).json({ error: 'No token' });
  try {
    req.jwtPayload = jwt.verify(token, process.env.JWT_SECRET!);
    next();
  } catch {
    return res.status(401).json({ error: 'Invalid token' });
  }
}
```

```javascript
import Papa from 'papaparse';

export function parseMenuCsv(csvText) {
  return new Promise((resolve, reject) => {
    Papa.parse(csvText, {
      header: true,
      complete: (results) => {
        const items = results.data
          .filter(row => row.name && row.price)
          .map(row => ({
            n: row.name,
            p: parseFloat(row.price),
            c: row.category || 'General',
            t: row.diet || 'Non-Veg',
            menuType: row.type?.toUpperCase() === 'LIQUOR' ? 'LIQUOR' : 'FOOD',
            printerTarget: row.type?.toUpperCase() === 'LIQUOR' ? 'BAR' : 'KITCHEN',
          }));
        resolve(items);
      },
      error: reject,
    });
  });
}
```

```typescript
import net from 'net';

export async function printToNetworkPrinter(ip: string, port: number = 9100, commands: Buffer): Promise<void> {
  return new Promise((resolve, reject) => {
    const socket = new net.Socket();
    const timeout = setTimeout(() => {
      socket.destroy();
      reject(new Error(`Printer at ${ip}:${port} timed out`));
    }, 5000);

    socket.connect(port, ip, () => {
      socket.write(commands);
      socket.end();
      clearTimeout(timeout);
      resolve();
    });
    socket.on('error', (err) => { clearTimeout(timeout); reject(err); });
  });
}
```
